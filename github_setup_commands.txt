# GitHub Setup Commands
# Run these commands after creating your GitHub repository

# 1. Add remote origin (replace YOUR_USERNAME with your actual GitHub username)
git remote add origin https://github.com/YOUR_USERNAME/mcp-travel-planning-system.git

# 2. Rename branch to main (GitHub's default)
git branch -M main

# 3. Push to GitHub
git push -u origin main

# 4. Verify the push worked
git remote -v

# Alternative: If you want to use SSH instead of HTTPS
# git remote add origin git@github.com:YOUR_USERNAME/mcp-travel-planning-system.git

# Note: Replace YOUR_USERNAME with your actual GitHub username
# Example: https://github.com/johndoe/mcp-travel-planning-system.git
